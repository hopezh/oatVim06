fig = px.line_mapbox(

    lon="d_e_longitude",
    color="h_b_int_num_id",
    zoom=3,
    height=1200,
    # animation_frame=df["h_b_int_num_id"]  # option 1
    # animation_frame="h_b_int_num_id"      # option 2

    # mapbox_style="open-street-map",

    # mapbox_zoom=4,

    margin={"r":0,"t":0,"l":0,"b":0},
)

fig.show()
